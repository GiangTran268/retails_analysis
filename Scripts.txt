-- Create table products_summary
SELECT
    retails.products.product_key,
    retails.products.product_name,
    retails.products.brand,
    retails.products.subcategory,
    retails.products.category,
    
    -- Units sold
    SUM(retails.sales.quantity) AS number_of_goods_sold,
    
    -- Revenue per product
    SUM(retails.sales.quantity * retails.products.unit_price_usd) AS revenue_of_goods,
    
    -- Profit per product
    SUM(retails.sales.quantity * (retails.products.unit_price_usd - retails.products.unit_cost_usd)) AS profit,
    
    -- Profit margin
    SUM(retails.sales.quantity * (retails.products.unit_price_usd - retails.products.unit_cost_usd)) 
    / SUM(retails.sales.quantity * retails.products.unit_price_usd) AS profit_margin

FROM retails.products
LEFT JOIN retails.sales
    ON retails.products.product_key = retails.sales.product_key
GROUP BY
    retails.products.product_key,
    retails.products.product_name,
    retails.products.brand,
    retails.products.subcategory,
    retails.products.category;

-- Create table customers_summary
SELECT 
    retails.customers.customer_key,
    retails.customers.name
    retails.customers.gender,
    retails.customers.city,
    retails.customers.state,
    retails.customers.country,
    
    -- Calculate age
    DATEDIFF(YEAR, retails.customers.birthday, GETDATE()) AS age,
    
     -- Count distinct orders
    COUNT(DISTINCT retails.sales.order_number) AS number_of_orders,
    
    -- Calculate Revenue per Customer
    SUM(retails.sales.quantity * retails.products.unit_price_usd) AS total_revenue,
    
    -- Average Order Value (AOV)
    SUM(retails.sales.quantity * retails.products.unit_price_usd) / COUNT(DISTINCT retails.sales.order_number) AS avg_order_value

FROM retails.customers
LEFT JOIN retails.sales
    ON retails.customers.customer_key = retails.sales.customer_key    
LEFT JOIN retails.products
	ON retails.sales.product_key = retails.products.product_key 

GROUP BY 
    retails.customers.customer_key,
    retails.customers.gender,
    retails.customers.city,
    retails.customers.state,
    retails.customers.country,
    retails.customers.birthday;

-- Calculate revenue and profit per month and per year
SELECT
    YEAR(retails.sales.order_date) AS order_year,
    MONTH(retails.sales.order_date) AS order_month,
    DATENAME(MONTH, retails.sales.order_date) AS month_name,

    SUM(retails.sales.quantity * retails.products.unit_price_usd) AS total_revenue,
    SUM(retails.sales.quantity * (retails.products.unit_price_usd - retails.products.unit_cost_usd)) AS total_profit

FROM retails.sales
LEFT JOIN retails.products
    ON retails.sales.product_key = retails.products.product_key

GROUP BY
    YEAR(retails.sales.order_date),
    MONTH(retails.sales.order_date),
    DATENAME(MONTH, retails.sales.order_date)

ORDER BY
    order_year,
    order_month;
